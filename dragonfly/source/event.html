     * @return {HTMLElement}
     */

        return element;
    };
    
    /**
     * 移除事件
     * 
     * @param {HTMLElement} element

     * @return {HTMLElement}
     */
    event.off = function(element, type, callback) {
        
        return element;
    };
    
    /**
     * 触发事件
     * 
     * @param {HTMLElement} element
     * @param {String} type
     * @return {HTMLElement}
     */

        return element;
    };
    
    return event;
});
